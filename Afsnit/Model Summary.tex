\section{Model Summary}

\subsection{Key equations}

Table \ref{tab:model_formulation} summarises the main structural equations of the model across all specifications.

\begin{table}[H]
\centering
\renewcommand{\arraystretch}{1.6}
\begin{tabularx}{\textwidth}{lX}
\hline
\textbf{Component} & \textbf{Equation} \\
\hline
Period utility & $\displaystyle U_t = \frac{c_t^{1+\eta}}{1+\eta} - \beta(n_t)\frac{h_t^{1+\gamma}}{1+\gamma}$ \\
Labour disutility & $\beta(n_t) = \beta_0 + \beta_1\, n_t$ \\
Wage function & $w_t(k_t) = (1-\tau)\,w\,(1 + \alpha\, k_t)$ \\
Human capital & $k_{t+1} = k_t + h_t$ \\
Fertility & $n_{t+1} = n_t + 1$ with prob.\ $p_{\text{birth}}$; otherwise $n_{t+1} = n_t$ \\
\hline
Budget constraint (baseline) & $a_{t+1} = (1+r)\bigl(a_t + (1-\tau)w_t h_t - c_t\bigr)$ \\
\quad + Spouse income (Q2) & $a_{t+1} = (1+r)\bigl(a_t + (1-\tau)(w_t h_t + y_t) - c_t\bigr)$,\quad $y_t = y_0 + y_1 t$ \\
\quad + Childcare cost (Q3) & $a_{t+1} = (1+r)\bigl(a_t + (1-\tau)(w_t h_t + y_t) - \theta n_t - c_t\bigr)$ \\
\quad + Stochastic spouse (Q5) & $a_{t+1} = (1+r)\bigl(a_t + (1-\tau)(w_t h_t + s_t y_t) - c_t\bigr)$,\quad $s_t \in \{0,1\}$ \\
\hline
\end{tabularx}
\caption{Key model equations across all specifications.}
\label{tab:model_formulation}
\end{table}
